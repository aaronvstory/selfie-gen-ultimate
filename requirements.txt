# ALL three opencv variants capped <4.12: opencv 4.12+ declares numpy>=2
# (Python 3.9+), which conflicts with the numpy<2 pin below (required by
# tensorflow 2.16.2). Without these caps a fresh install resolves opencv 4.13
# -> numpy>=2 vs numpy<2 -> pip ResolutionImpossible (or a fragile backtrack).
# opencv-python is pulled transitively by deepface, opencv-contrib-python by
# mediapipe (matplotlib drawing_utils); pin both explicitly so the resolution
# is deterministic, not backtrack-dependent. Mirrors the opencv-contrib-python
# <4.12 cap already in setup_macos.sh (same root cause).

opencv-python>=4.5.5.64,<4.12
opencv-contrib-python<4.12
mediapipe==0.10.35
deepface==0.0.92
# TensorFlow MUST be pinned explicitly. deepface==0.0.92 only declares a loose
# `tensorflow>=1.9.0`, so on a fresh install pip resolves whatever is newest
# (e.g. 2.21.x) and only *backtracks* to 2.16.2 because tf-keras==2.16.0
# constrains it — a fragile resolution that varies by pip version / platform.
# When pip lands a TF != 2.16.2, the every-launch dependency_health_check
# (REPAIR_PACKAGES expects exactly 2.16.2 + tensorflow-intel==2.16.2 on win32)
# fails and forces a repair cycle on first launch; on a slow/firewalled
# connection that repair can hard-fail and the launcher exits before the GUI.
# Pinning here (mirrors requirements-hashed.txt + REPAIR_PACKAGES) makes the
# first install land the exact stack the health probe expects — no repair loop.
# numpy<2 is required: TF 2.16.2's ml-dtypes~=0.3.1 needs numpy 1.26.x; an
# unpinned numpy 2.x breaks TF import.
tensorflow==2.16.2
tensorflow-intel==2.16.2; sys_platform == "win32"
numpy>=1.26,<2
tf-keras==2.16.0
retina-face==0.0.17
# absl-py MUST be pinned explicitly even though tensorflow + mediapipe both
# declare it. The rPPG launcher installs mediapipe with `--no-deps` (to stop it
# pulling numpy 2.x), which SKIPS mediapipe's `absl-py~=2.3`. And on a fresh
# install the TF-side absl can resolve as "already satisfied" against a global
# Python or simply not land in the venv — leaving absl absent. rPPG's
# rppg_injector.py does an UNGUARDED `import absl.logging` at module top, so a
# missing absl crashes the whole rPPG step (the friend's v2.13 "rPPG fails,
# everything else works" bug). Pin it so a fresh venv always gets it. The
# ~=2.3 range matches what mediapipe 0.10.35 + TF 2.16.2 expect.
absl-py~=2.3
# torch is required by DeepFace's anti-spoofing classifier (MiniFASNetV2).
# Without it the engine fails ALL similarity comparisons when anti_spoofing=True.
# This installs the DEFAULT (CPU/PyPI) wheel; the launchers then run
# `gpu_bootstrap.py --select-torch` which detects the GPU and reinstalls the
# hardware-appropriate build (Windows+NVIDIA -> CUDA index, macOS -> default
# MPS/CPU, else -> CPU index). v2.17: no longer a manual swap. CUDA only
# affects speed, not correctness, so the CPU wheel here is always safe.
torch>=2.2,<3
